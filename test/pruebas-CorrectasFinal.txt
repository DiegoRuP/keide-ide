// --- Declaración de Función ---
// Calcula el área y la multiplica por un factor.
float calculate_area(float width, float height, int factor) {
    float base_area = width * height;
    float result = base_area * factor;
    
    cout << "Calculando área... Resultado parcial: ";
    cout << result;
    
    return result;
}
