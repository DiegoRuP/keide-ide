// Prueba de FOR y SWITCH (Sintaxis Corregida)

main {
    int opcion = 2;
    string resultado = "ninguno";

    // Prueba de Switch con 'end'
    switch (opcion)
        case 1:
            resultado = "uno";
            break;
        case 2:
            resultado = "dos";
            break;
        case 3:
            resultado = "tres";
            break;
        default:
            resultado = "otro";
            break;
    end

    cout << "El resultado del switch es: " + resultado;

    // Prueba de For con 'end' y sin el último ';'
    int acumulador = 0;
    for (int i = 0; i < 5; i = i + 1)
        acumulador = acumulador + i;
        cout << "Iteración del for...";
    end
    
    cout << "El resultado del for es: " + acumulador;
}
