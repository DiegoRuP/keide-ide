main {
    // --- Declaración y Asignación de Variables ---
    int score = 0;

    // --- Uso de Estructuras Condicionales (if) ---

        cout << "El jugador está en buena condición.";

    int counter = 3;
    while (counter > 0)
        cout << "Tiempo restante: " + counter;
        counter = counter - 1;
    end
    
    do
      cout << "Fin del bucle while.";
      counter = counter + 1;
    until (counter == 1);

    // --- Sombreado de Variables (Shadowing) ---

    string game_status = "finished";
    if (game_status == "finished") then
        cout << "El juego ha terminado.";
    end
    
    // --- Validación de Ambito ---
