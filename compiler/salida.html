main {
    // Declaraciones
    int a, b, c;
    float x, y;
    string saludo, nombre;

    // Asignaciones
    a = 10;
    b = 20;
    x = 3.14;
    saludo = "Hola ";
    nombre = "Mundo";

    // Operaciones y asignaciones
    c = a + (b * 2); // c = 50
    y = c / x;     // y = 50 / 3.14 (asignación de int a float es válida)
    
    string mensaje;
    mensaje = saludo + nombre; // Concatenación de strings

    // Salida
    cout << "El valor final de c es: ";
    cout << c;

    // Estructuras de control
    if (a < b) then
        cout << "a es menor que b";
    end

    while (a < 15) then
        a = a + 1;
    end
    
    cout << "El nuevo valor de a es: ";
    cout << a;
}
